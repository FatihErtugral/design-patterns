Obersever Design Pattern


    
    


    em.on('open', (data) => {
        const out = document.querySelector('.oberver-event-manager-out');
    
        document.body.style.backgroundColor = 'purple';
    
        if (out) {
            out.innerHTML += data;
        }
    });
    em.on('close', (data) => {
        const out = document.querySelector('.oberver-event-manager-out');
    
        document.body.style.backgroundColor = 'darkgray';
    
        if (out) {
            out.innerHTML += '\n' + data;
        }
    });
    
    setTimeout(() => {
        em.emit('open', 'open event triggered');
    }, 1000);
    
    setTimeout(() => {
        em.emit('close', 'close event triggered');
    }, 2000);
